import { useState, useEffect } from "react";
import AccountHeader from "@/components/account/AccountHeader";
import AccountTabs from "@/components/account/AccountTabs";
import ProfileSettings from "@/components/account/ProfileSettings";
import { UserAccount } from "@/types";
import { getCurrentUserAccount, updateUserProfile } from "@/lib/mockData";

type AccountTab =
  | "profile"
  | "subscription"
  | "billing"
  | "security"
  | "notifications";

export default function AccountPage() {
  const [account, setAccount] = useState<UserAccount | null>(null);
  const [activeTab, setActiveTab] = useState<AccountTab>("profile");
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    loadAccount();
  }, []);

  const loadAccount = async () => {
    try {
      const userAccount = await getCurrentUserAccount();
      setAccount(userAccount);
    } catch (error) {
      console.error("Erreur lors du chargement du compte:", error);
    } finally {
      setLoading(false);
    }
  };

  const handleUpdateProfile = async (updates: Partial<UserAccount>) => {
    if (!account) return;

    try {
      await updateUserProfile(updates);
      setAccount((prev) => (prev ? { ...prev, ...updates } : null));
    } catch (error) {
      console.error("Erreur lors de la mise √† jour:", error);
      throw error;
    }
  };

  const renderTabContent = () => {
    if (!account) return null;

    switch (activeTab) {
      case "profile":
        return (
          <ProfileSettings account={account} onUpdate={handleUpdateProfile} />
        );

      case "subscription":
        return (
          <div className="bg-white rounded-lg border border-gray-200 p-6">
            <h3 className="text-lg font-semibold text-gray-900 mb-4">
              Gestion de l'abonnement
            </h3>
            <p className="text-gray-600">
              Contenu de l'abonnement (sera impl√©ment√© dans ACCOUNT-02)

            <h3 className="text-lg font-semibold text-gray-900 mb-4">
              Historique de facturation
            </h3>
            <p className="text-gray-600">
              Historique des factures (sera impl√©ment√© dans ACCOUNT-03)
            </p>
          </div>
        );

      case "security":
        return (
          <div className="bg-white rounded-lg border border-gray-200 p-6">
            <h3 className="text-lg font-semibold text-gray-900 mb-4">
              S√©curit√© et confidentialit√©

            <p className="text-gray-600">
              Param√®tres de s√©curit√© (sera impl√©ment√© dans une future version)
            </p>
          </div>
        );

      case "notifications":
        return (
          <div className="bg-white rounded-lg border border-gray-200 p-6">
            <h3 className="text-lg font-semibold text-gray-900 mb-4">
              Pr√©f√©rences de notifications
            </h3>
            <p className="text-gray-600">
              Param√®tres de notifications (sera impl√©ment√© dans une future
